CREATE TABLE STUDENT_COURSE (
	STUDENT_ID BIGINT(10) NOT NULL,
	COURSE_ID BIGINT(10) NOT NULL,
	TERM VARCHAR(20) NOT NULL,
	FINAL_GRADE DOUBLE(4,2),
	CONSTRAINT STUDENT_COURSE_ID PRIMARY KEY (STUDENT_ID, COURSE_ID, TERM)
);

CREATE TABLE STUDENT_INFORMATION (
	STUDENT_ID BIGINT(10) NOT NULL UNIQUE,
	LAST_NAME VARCHAR(100),
	FIRST_NAME VARCHAR(100),
	DATE_OF_BIRTH DATETIME,
	ADMITTED_DATE DATETIME,
	SEX VARCHAR(1),
	PRIMARY KEY(STUDENT_ID)
);

CREATE TABLE STUDENT_ADDRESS (
	STUDENT_ID BIGINT(10) NOT NULL UNIQUE,
	NUMBER INT(10) NOT NULL,
	STREET_ADDRESS VARCHAR(100) NOT NULL,
	CITY VARCHAR(100) NOT NULL,
	STATE_OR_PROVINCE VARCHAR(100) NOT NULL,
	COUNTRY VARCHAR(100) NOT NULL,
	ZIP_CODE INT(5) NOT NULL,
	CONSTRAINT STUDENT_ADDRESS_ID PRIMARY KEY (STUDENT_ID, NUMBER, STREET_ADDRESS, CITY, STATE_OR_PROVINCE, COUNTRY, ZIP_CODE)
);

CREATE TABLE DEPARTMENT (
	DEPARTMENT_ID BIGINT(10) NOT NULL UNIQUE,
	DEPARTMENT_TITLE VARCHAR(100),
	DEPARTMENT_DESCRIPTION TEXT,
	PRIMARY KEY(DEPARTMENT_ID)
);

CREATE TABLE COURSE_CATALOG (
	COURSE_ID BIGINT(10) NOT NULL UNIQUE,
	ASSIGNED_INSTRUCTOR_ID BIGINT(10),
	REQUIREMENT_ID BIGINT(10),
	LOCATION_ID BIGINT(10),
	DEPARTMENT_ID BIGINT(10),
	TITLE VARCHAR(100),
	DESCRIPTION TEXT,
	PRIMARY KEY(COURSE_ID)
);

CREATE TABLE COURSE_REQUIREMENTS (
	REQUIREMENT_ID BIGINT(10) NOT NULL UNIQUE,
	REQUIREMENT_TITLE VARCHAR(100),
	REQUIREMENT_DESCRIPTION TEXT,
	PRIMARY KEY(REQUIREMENT_ID)
);

CREATE TABLE LOCATION (
	LOCATION_ID BIGINT(10) NOT NULL UNIQUE,
	LOCATION_NAME VARCHAR(100),
	MAX_CAPACITY INT(5),
	PRIMARY KEY(LOCATION_ID)
);

CREATE TABLE STUDENT_GRADE_YEAR (
	STUDENT_ID BIGINT(10) NOT NULL UNIQUE,
	GRADE_YEAR INT(2) NOT NULL,
	CONSTRAINT STUDENT_GRADE_YEAR_ID PRIMARY KEY (STUDENT_ID, GRADE_YEAR)
);

CREATE TABLE STUDENT_ATTENDANCE (
	STUDENT_ID BIGINT(10) NOT NULL,
	DATE DATETIME NOT NULL,
	ABSENCE_ID BIGINT(10),
	CONSTRAINT STUDENT_ATTENDANCE_ID PRIMARY KEY (STUDENT_ID, DATE)
);

CREATE TABLE STAFF_INFORMATION (
	STAFF_ID BIGINT(10) NOT NULL UNIQUE,
	ROLE_ID BIGINT(10),
	LAST_NAME VARCHAR(100),
	FIRST_NAME VARCHAR(100),
	GOVERNMENT_ID INT(10),
	SALARY DECIMAL(10, 2),
	PHONE_NUMBER VARCHAR(19),
	DATE_OF_BIRTH DATETIME,
	ADMITTED_DATE DATETIME,
	PRIMARY KEY(INSTRUCTOR_ID)
);

CREATE TABLE STAFF_ROLE (
	ROLE_ID BIGINT(10) NOT NULL UNIQUE,
	ROLE_TITLE VARCHAR(100),
	ROLE_DESCRIPTION TEXT,
	PRIMARY KEY(ROLE_ID)
);

CREATE TABLE ABSENCE_TYPES (
	ABSENCE_ID BIGINT(10) NOT NULL UNIQUE,
	ABSENCE_TITLE VARCHAR(100),
	DESCRIPTION TEXT,
	PRIMARY KEY(ABSENCE_ID)
);

CREATE TABLE STAFF_ADDRESS (
	STAFF_ID BIGINT(10) NOT NULL UNIQUE,
	NUMBER INT(10) NOT NULL,
	STREET_ADDRESS VARCHAR(100) NOT NULL,
	CITY VARCHAR(100) NOT NULL,
	STATE_OR_PROVINCE VARCHAR(100) NOT NULL,
	COUNTRY VARCHAR(100) NOT NULL,
	ZIP_CODE INT(5) NOT NULL,
	CONSTRAINT STUDENT_ADDRESS_ID PRIMARY KEY (STAFF_ID, NUMBER, STREET_ADDRESS, CITY, STATE_OR_PROVINCE, COUNTRY, ZIP_CODE)
);
