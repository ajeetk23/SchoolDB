CREATE VIEW VW_STUDENT_OVERVIEW_BY_HEADMASTER AS (
	SELECT S.STUDENT_ID, S.FIRST_NAME, S.LAST_NAME, G.GRADE_YEAR FROM STUDENT_INFORMATION S
	LEFT JOIN STUDENT_GRADE_YEAR G
	ON S.STUDENT_ID = G.STUDENT_ID
);

CREATE VIEW VW_STAFF_OVERVIEW_BY_HEADMASTER AS (
	SELECT S.STAFF_ID, S.FIRST_NAME, S.LAST_NAME, R.ROLE_TITLE FROM STAFF_INFORMATION S
	LEFT JOIN STAFF_ROLE R
	ON S.ROLE_ID = R.ROLE_ID
);

CREATE VIEW VW_CLASS_ROSTER AS (
	SELECT C.TITLE, SC.TERM, S.FIRST_NAME, S.LAST_NAME FROM COURSE_CATALOG C
    INNER JOIN STUDENT_COURSE SC
    ON SC.COURSE_ID = C.COURSE_ID
    INNER JOIN STUDENT_INFORMATION S
    ON SC.STUDENT_ID = S.STUDENT_ID
);

CREATE VIEW VW_STUDENT AS (
	SELECT SC.STUDENT_ID, SC.COURSE_ID FROM STUDENT_COURSE SC
);
